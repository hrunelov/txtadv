<command>     ::= <movement>
                | <action>

<movement>    ::= [go | move | walk] <direction>

<direction>   ::= <east>
                | <south>
                | <west>
                | <north>
                | <up>
                | <down>

<east>        ::= east | e
<south>       ::= south | s
<west>        ::= west | w
<north>       ::= north | n
<up>          ::= up | u

<action>      ::= <verb> <noun> [with <noun>]

<verb>        ::= <examine>
                | <pickup>
                | <engage>
                | <custom_verb>
                
<examine>     ::= examine | check
<pickup>      ::= pickup | take
<engage>      ::= engage | attack

<noun>        ::= <location>
                | <inventory>
                | equipment
                | <custom_noun>

<location>    ::= location | area
<inventory>   ::= inventory | items
